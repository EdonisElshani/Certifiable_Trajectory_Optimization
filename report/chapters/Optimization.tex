\subsection{Control of the Acrobot with the Kinematic-Substituted Reduced Model}
\label{sec:control_acrobot_reduced_model2}

The second reduced formulation eliminates the translational position and velocity
variables by reconstructing the center-of-mass positions from the rotations. The
finite-difference stencil is then rewritten using the Lie group kinematics
\[
    R_{i,k+1}=R_{i,k}F_{i,k},
    \qquad
    R_{i,k-1}=R_{i,k}F_{i,k-1}^{\top}.
\]
Hence, the translational dynamics depend only on
\(R_{i,k}\), \(F_{i,k-1}\), and \(F_{i,k}\). The decision variables are
\[
\left\{
R_{1,k},R_{2,k}
\right\}_{k=0}^{N},
\qquad
\left\{
F_{1,k},F_{2,k}
\right\}_{k=0}^{N-1},
\]
together with the normalized Lagrange multiplier and control variables
\[
\left\{
\hat\lambda_{0,k},\hat\lambda_{12,k},\hat u_k
\right\}_{k=1}^{N-1}.
\]
The physical control and multiplier values used in the dynamics are obtained by
\[
u_k^{\mathrm{phys}}
=
u_{\max}\hat u_k,
\qquad
\bar{\lambda}_{0,k}
=
\lambda_{\max}\hat\lambda_{0,k},
\qquad
\bar{\lambda}_{12,k}
=
\lambda_{\max}\hat\lambda_{12,k}.
\]
Here, \(\lambda_{\max}\) denotes the effective physical multiplier scale after
absorbing the previous scaling factor. Thus, the old combination
\(s_\lambda\lambda_{\max}^{\mathrm{old}}\) is replaced by a single
\(\lambda_{\max}\).

\label{eq:forced_acrobot_reduced_model2_scalar_trans1x}
\\
-
m_1\frac{l_1}{2}\widehat{\Delta c}_{1,k}
+
\frac{h}{2}m_1g
-
\frac{h}{2}
\lambda_{\max}
\left(
\hat\lambda_{0,k}^{(2)}
+
\hat\lambda_{12,k}^{(2)}
\right)
&=0,
\label{eq:forced_acrobot_reduced_model2_scalar_trans1y}
\\
m_2
\left(
l_1\widehat{\Delta s}_{1,k}
+
\frac{l_2}{2}\widehat{\Delta s}_{2,k}
\right)
+
\frac{h}{2}
\lambda_{\max}
\hat\lambda_{12,k}^{(1)}
&=0,
\label{eq:forced_acrobot_reduced_model2_scalar_trans2x}

-
m_2
\left(

+
\frac{l_2}{2}\widehat{\Delta c}_{2,k}
\right)
+
\frac{h}{2}m_2g
+
\frac{h}{2}
\lambda_{\max}
\hat\lambda_{12,k}^{(2)}
&=0,
\label{eq:forced_acrobot_reduced_model2_scalar_trans2y}
\end{align}
for \(k=1,\ldots,N-1\).

The scalar rotational dynamics are
\begin{align}
(\delta_{1,1}+\delta_{1,2})
(b_{1,k-1}-b_{1,k})
+
h^2
\left(
\mu_{1,0,k}
+
\mu_{1,12,k}
-
u_{\max}\hat u_k
\right)
&=0,
\label{eq:forced_acrobot_reduced_model2_scalar_rot1}
\\
(\delta_{2,1}+\delta_{2,2})
(b_{2,k-1}-b_{2,k})
+
h^2
\left(
u_{\max}\hat u_k
-
\mu_{2,12,k}
\right)
&=0,
\label{eq:forced_acrobot_reduced_model2_scalar_rot2}
\end{align}
where
\begin{align}
\mu_{1,0,k}
&=
\rho_{1,0}^{(1)}
\left(
-s_{1,k}\bar{\lambda}_{0,k}^{(1)}
+
c_{1,k}\bar{\lambda}_{0,k}^{(2)}
\right)
-
\rho_{1,0}^{(2)}
\left(
c_{1,k}\bar{\lambda}_{0,k}^{(1)}
+
s_{1,k}\bar{\lambda}_{0,k}^{(2)}
\right),
\\
\mu_{1,12,k}
&=
\rho_{1,12}^{(1)}
\left(
-s_{1,k}\bar{\lambda}_{12,k}^{(1)}
+
c_{1,k}\bar{\lambda}_{12,k}^{(2)}
\right)
-
\rho_{1,12}^{(2)}
\left(
c_{1,k}\bar{\lambda}_{12,k}^{(1)}
+
s_{1,k}\bar{\lambda}_{12,k}^{(2)}
\right),
\\
\mu_{2,12,k}
&=

\left(
-s_{2,k}\bar{\lambda}_{12,k}^{(1)}
+

\right)
-

\left(

+

\end{align}
Here,
\[
\bar{\lambda}_{0,k}
=
\lambda_{\max}\hat\lambda_{0,k},
\qquad
\bar{\lambda}_{12,k}
=
\lambda_{\max}\hat\lambda_{12,k}.

The scalar kinematic equations are
\begin{align}
c_{i,k}
-
c_{i,k-1}a_{i,k-1}
+
s_{i,k-1}b_{i,k-1}
&=0,
\\
s_{i,k}
-
s_{i,k-1}a_{i,k-1}
-
c_{i,k-1}b_{i,k-1}
&=0,
\end{align}
for \(i=1,2\) and \(k=1,\ldots,N\).

The \(SO(2)\) constraints are
\begin{align}
c_{i,k}^2+s_{i,k}^2-1 &=0,
\qquad
i=1,2,\quad k=0,\ldots,N,
\\
a_{i,k}^2+b_{i,k}^2-1 &=0,
\qquad
i=1,2,\quad k=0,\ldots,N-1.
\end{align}

The inequality constraints are
\begin{align}
a_{i,k}-a_{i,\min} &\ge 0,
\qquad
i=1,2,\quad k=0,\ldots,N-1,
\\
1-\hat u_k^2 &\ge 0,
\qquad
k=1,\ldots,N-1,
\\
1-\left(\hat\lambda_{0,k}^{(j)}\right)^2 &\ge 0,
\qquad
j=1,2,\quad k=1,\ldots,N-1,
\\
1-\left(\hat\lambda_{12,k}^{(j)}\right)^2 &\ge 0,
\qquad
j=1,2,\quad k=1,\ldots,N-1.
\end{align}
