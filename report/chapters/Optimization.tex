\begin{subequations}\label{eq:forced_3d_pendulum_pop}
\begin{align}
\min_{\substack{\{R_k\}_{k=0}^{N}\\[0.1em]\{F_k\}_{k=0}^{N-1},\,\{u_{p,k}\}_{k=1}^{N-1}}}
\quad
&
\rho_R \left\| R_N - R_N^{\mathrm{des}} \right\|_F^2
+
\rho_F \left\| F_{N-1} - F^{\mathrm{des}} \right\|_F^2
\notag\\
&\quad
+
\sum_{k=0}^{N-1}
\alpha_R \left\| R_k - R_N^{\mathrm{des}} \right\|_F^2
+
\sum_{k=0}^{N-2}
\alpha_F \left\| F_k - F^{\mathrm{des}} \right\|_F^2
\notag\\
&\quad
+
\sum_{k=1}^{N-1}
\gamma^{-1}\left\|u_{p,k}\right\|_2^2 .
\label{eq:forced_3d_pendulum_pop_obj}
\\[0.4em]
\text{s.t.}\quad
&
\delta_2 f_{32,k+1}-\delta_3 f_{23,k+1}
-\delta_3 f_{32,k}+\delta_2 f_{23,k}
-h^2\!\left(M_{1,k+1}+u_{1,k+1}\right)=0,
\label{eq:forced_3d_pendulum_pop_dyn1}
\\
&
\delta_3 f_{13,k+1}-\delta_1 f_{31,k+1}
-\delta_1 f_{13,k}+\delta_3 f_{31,k}
-h^2\!\left(M_{2,k+1}+u_{2,k+1}\right)=0,
\label{eq:forced_3d_pendulum_pop_dyn2}
\\
&
\delta_1 f_{21,k+1}-\delta_2 f_{12,k+1}
-\delta_2 f_{21,k}+\delta_1 f_{12,k}
-h^2\!\left(M_{3,k+1}+u_{3,k+1}\right)=0,
\label{eq:forced_3d_pendulum_pop_dyn3}
\\
&
(R_{k+1})_{ij}-\sum_{\ell=1}^{3}(R_k)_{i\ell}(F_k)_{\ell j}=0,
\label{eq:forced_3d_pendulum_pop_kin}
\\
&
\|r_{1,k}\|_2^2-1=\|r_{2,k}\|_2^2-1=\|r_{3,k}\|_2^2-1=0,
\label{eq:forced_3d_pendulum_pop_R1}
\\
&
r_{1,k}^\top r_{2,k}=r_{2,k}^\top r_{3,k}=r_{1,k}^\top r_{3,k}=0,
\label{eq:forced_3d_pendulum_pop_R2}
\\
&
r_{1,k}\times r_{2,k}-r_{3,k}
=
r_{2,k}\times r_{3,k}-r_{1,k}
=
r_{3,k}\times r_{1,k}-r_{2,k}
=
0_{3\times1},
\label{eq:forced_3d_pendulum_pop_R3}
\\
&
\|f_{1,k}\|_2^2-1=\|f_{2,k}\|_2^2-1=\|f_{3,k}\|_2^2-1=0,
\label{eq:forced_3d_pendulum_pop_F1}
\\
&
f_{1,k}^\top f_{2,k}=f_{2,k}^\top f_{3,k}=f_{1,k}^\top f_{3,k}=0,
\label{eq:forced_3d_pendulum_pop_F2}
\\
&
f_{1,k}\times f_{2,k}-f_{3,k}
=
f_{2,k}\times f_{3,k}-f_{1,k}
=
f_{3,k}\times f_{1,k}-f_{2,k}
=
0_{3\times1},
\label{eq:forced_3d_pendulum_pop_F3}
\\
&
\frac{\operatorname{tr}(F_k)-1}{2}\ge f_{c,\min},
\label{eq:forced_3d_pendulum_pop_step}
\\
&
u_{\max}^2-\|u_{p,k}\|_2^2\ge 0.
\label{eq:forced_3d_pendulum_pop_u}
\end{align}
\end{subequations}

In \eqref{eq:forced_3d_pendulum_pop_dyn1}--\eqref{eq:forced_3d_pendulum_pop_dyn3},
the discrete Euler--Lagrange constraints hold for $k=0,\dots,N-2$.
The kinematic constraint \eqref{eq:forced_3d_pendulum_pop_kin} holds for
$i,j=1,2,3$ and $k=0,\dots,N-1$.
The constraints \eqref{eq:forced_3d_pendulum_pop_R1}--\eqref{eq:forced_3d_pendulum_pop_R3}
hold for $k=0,\dots,N$, while
\eqref{eq:forced_3d_pendulum_pop_F1}--\eqref{eq:forced_3d_pendulum_pop_F3},
\eqref{eq:forced_3d_pendulum_pop_step}, and
\eqref{eq:forced_3d_pendulum_pop_u} hold for $k=0,\dots,N-1$.
